74 class s_buffer
75 {
76 private:
77  char* d_buffer;
78  size_t d_n_used, d_n_alloc;
79 
80 public:
81  inline s_buffer (size_t n_alloc = 0) {
82  d_n_used = 0;
83  d_n_alloc = n_alloc;
84  if (n_alloc) {
85  d_buffer = (char*) new char [n_alloc];
86  } else {
87  d_buffer = 0;
88  }
89  };
90  inline ~s_buffer () {
91  if (d_n_alloc) {
92  delete [] d_buffer;
93  }
94  };
95  inline size_t n_used () { return (d_n_used); };
96  inline void n_used (size_t bufLen) {
97  d_n_used = (bufLen > d_n_alloc) ? d_n_alloc : bufLen; };
98  inline size_t n_alloc () { return (d_n_alloc); };
99  void buffer (char* l_buffer, size_t bufLen) {
100  if (bufLen > d_n_alloc) {
101  std::cerr << "s_buffer::set: Copying only allocated bytes." << std::endl;
102  bufLen = d_n_alloc;
103  }
104  if (!l_buffer) {
105  std::cerr << "s_buffer::set: NULL buffer." << std::endl;
106  return;
107  }
108  bcopy (l_buffer, d_buffer, bufLen);
109  d_n_used = bufLen;
110  };
111  inline char* buffer () { return (d_buffer); };
112  inline void reset () {
113  bzero (d_buffer, d_n_alloc);
114  d_n_used = 0;
115  };
116 };
